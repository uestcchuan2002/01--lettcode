## 1.两数之和
解决思路：
1. 直接两个for循环，暴力解决
```cpp
vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> ans;
        for (int i = 0; i < nums.size() - 1; i++) {
            for (int j = i + 1; j < nums.size(); j++) {
                if (nums[i] + nums[j] == target) {
                    ans.push_back(i);
                    ans.push_back(j);
                    break;
                }
            }
        }
        return ans;
    }
```
2. 采用哈希表查找
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> m_map;
        for (int i = 0; ; i++) {
            auto it = m_map.find(target - nums[i]);
            if (it != m_map.end()) {
                return {it->second, i};
            }
            m_map[nums[i]] = i;
        }
    }
};
```


**排序方法:**
1.对于数字排序
```cpp
#include <iostream>
#include <algorithm>

int main {
    // 1. 普通数组
    int arr[] = {5, 2, 9, 1, 5, 6};
    int arrSize = sizeof(arr) / sizeof(arr[0]);

    // 排序: std::sort(起始地址， 结束地址)
    std::sort(arr, arr + arrSize);

    // 输出结果：1 2 5 5 6 9
    std::cout << "数组升序排序结果：";
    for (int i = 0; i < arrSize; ++i) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;

    return 0;
}
```
2.对std::vector排序
```cpp
#include <iostream>
#include <algorithm>
#include <vector>

int main() {
    std::vector<int> vec = {8, 3, 7, 1, 9, 2};

    // 升序排序
    std::sort(vec.begin(), vec.end());

    // 输出结果：1 2 3 7 8 9
    std::cout << "vector升序排序结果：";
    for (int num : vec) { // 范围for循环（C++11+）
        std::cout << num << " ";
    }
    std::cout << std::endl;

    return 0;
}
```

## 49.字母异位词分组
**采用哈希表解决：**
```cpp
vector<vector<string>> groupAnagrams(vector<string>& strs) {
    unordered_map<string, vector<string>> m;
    // 采用引用的方式：避免拷贝字符串
    for (string &s : strs) {
        string s_old = s;
        ranges::sort(s_old);    // sort(s_old.begin(), s_old.end());
        m[s_old].push_back(s);
    }

    vector<vector<string>> ans;
    ans.reserve(m.size());
    for (auto& pair : m) {
        vector<string>& value = pair.second;
        ans.push_back(value);
    }
    return ans;
}
```
